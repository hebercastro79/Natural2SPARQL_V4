# TEMPLATE_4A -- Qual foi o volume negociado nas ações do setor <setor> em <data>?
SELECT ?O1 (SUM(?ANS) AS ?volumeTotal)
WHERE {
  ?S4 P7 "#NOME_SETOR#"@pt .
  ?S1 P9 ?S4 .
  ?S1 P1 ?SO1 .
  ?SO1 (P8 | (P2 / P8)) ?O1 .
  ?SO1 (P1)*/P3 ?SO2 .
  ?SO2 P5 ?S2 .
  ?S2 P6 "#DATA#"^^xsd:date .
  ?SO2 #VALOR_DESEJADO# ?ANS .
}
GROUP BY ?O1
ORDER BY ?O1
