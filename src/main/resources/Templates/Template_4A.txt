# TEMPLATE_4A -- Qual foi o volume negociado nas ações do Setor Financeiro em 05/05/2023?
SELECT ?O1 (SUM(?ANS) AS ?volume)
WHERE {
  ?S5 P18 "#NOME_SETOR#"@pt .
  ?S1 P9 ?S5 .
  ?S1 P1 ?S2 .
  ?S2 P19 ?O1 .
  ?S2 P3 ?S6 .
  ?S6 P4 ?S4 .
  ?S4 P6 "#DATA#"^^xsd:date .
  ?S6 #VALOR_DESEJADO# ?ANS .
}
GROUP BY ?O1
ORDER BY ?O1
