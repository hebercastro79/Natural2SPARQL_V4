#Template 1C
#Qual foi o preço máximo da {empresa} em {data}?
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX b3: <https://dcm.ffclrp.usp.br/lssb/stock-market-ontology#>

SELECT ?ANS WHERE {
  ?empresa rdfs:label "{empresa}"@pt .
  ?empresa b3:temAcao ?acao .
  ?acao b3:temPregao ?pregao .
  ?pregao b3:temData "{data}"^^xsd:date .
  ?pregao b3:precoMaximo ?ANS .
}
LIMIT 1
