# TEMPLATE_5B -- Busca métrica com filtro de TIPO, seguindo o caminho completo.
SELECT ?ANS WHERE {
    ?S1 P7 ?label .
    FILTER(REGEX(STR(?label), "#ENTIDADE_NOME#", "i"))
                      
    ?S1 P1 ?SO1 .
    ?SO1 P1 ?intermediario .
    ?intermediario P2 ?tickerCode .  # P2 = representadoPor (Intermediário -> Codigo)
    ?tickerCode b3:ticker ?O1 .   # Propriedade final que contém o literal do ticker

    ?intermediario P3 ?SO2 .
    ?SO2 P5 ?S2 .
    ?S2 P6 "#DATA#"^^xsd:date .
    ?SO2 #VALOR_DESEJADO# ?ANS .
    
    FILTER(REGEX(STR(?O1), "#REGEX_PATTERN#"))
}
