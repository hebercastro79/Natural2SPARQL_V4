# TEMPLATE_2A -- Qual foi o código de negociação da ação da Gerdau?
SELECT ?ANS WHERE {
    ?S1 rdfs:label "#ENTIDADE_NOME#"@pt . # Ponto de entrada
    ?S1 P3 ?SO2 .                         # empresa -> temValorMobiliario -> valorMobiliario
    ?SO2 P5 ?O2 .                         # valorMobiliario -> representadoPor -> codigoNegociacao
    ?O2 rdfs:label ?ANS .                 # Pega o label do codigo, que é o próprio ticker
}
